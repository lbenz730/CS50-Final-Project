def downword(word):
    for i in range(len(word)):
        # return false for non alphabetical characters
        if not str.isalpha(word[i]):
            return False
            
        # return false characters in wrong order
        if i < (len(word) - 1) and str.lower(word[i]) < str.lower(word[i + 1]):
            return False
    
    # return true only if downword
    return True
